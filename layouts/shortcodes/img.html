{{- /* shortcodes/img.html Displays images with optional captions and custom width. Usage: {{< img src="images/photo.jpg" >}} {{< img src="images/photo.jpg" width="600" >}} {{< img src="images/photo.jpg" caption="Description of image" >}} {{< img src="images/photo.jpg" width="700" caption="Photo caption" alt="Alt text" >}} Parameters: - src: Image path (required) - width: Max width in pixels (default: 800) - caption: Optional caption displayed below image - alt: Alt text for accessibility (defaults to caption if provided) Images are responsive and centered with rounded corners. */ -}} {{- $src := .Get "src" -}} {{- $width := .Get "width" | default "800" -}} {{- $caption := .Get "caption" -}} {{- $alt := .Get "alt" | default $caption | default "" -}}
